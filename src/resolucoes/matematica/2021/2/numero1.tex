

\[
\boxed{\textbf{Indique as soluções da equação } -|x - 2| + 6 = 2 \\
\\
\textbf{A.} \; x = 2 \vee x = 6
\quad\quad
\textbf{B.} \; x = -4 \vee x = 4
\quad\quad
\textbf{C.} \; x = 2
\quad\quad
\textbf{D.} \; x = -2 \vee x = 6
\quad\quad
\textbf{E.} \; x = 4}

\]







\[
\text{Primeiro isolamos o valor absoluto: } -|x - 2| + 6 = 2 \;\Rightarrow\; -|x - 2| = -4
\]





\[
\text{Ao multiplicar ambos os lados por } -1, \text{ obtemos } |x - 2| = 4
\]





\[
\text{O valor absoluto de uma expressão ser igual a 4 significa que } x - 2 = 4 \text{ ou } x - 2 = -4
\]





\[
\text{Se } x - 2 = 4, \text{ então } x = 6
\]





\[
\text{Se } x - 2 = -4, \text{ então } x = -2
\]




\[
\color{blue}{\large {\text{Solução: }\quad x = -2 \quad \text{ou} \quad x = 6}}
 \quad\quad  \]
