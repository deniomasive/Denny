% Enunciado + alternativas dentro da mesma caixa


\[
\boxed{
\begin{array}{l}
\textbf{Indique as soluções da equação } -|x - 2| + 6 = 2 \

\[10pt]
\textbf{A.} \; x = 2 \vee x = 6 \

\[6pt]
\textbf{B.} \; x = -4 \vee x = 4 \

\[6pt]
\textbf{C.} \; x = 2 \

\[6pt]
\textbf{D.} \; x = -2 \vee x = 6 \

\[6pt]
\textbf{E.} \; x = 4
\end{array}
}
\]



% Resolução detalhada argumentada


\[
\text{Primeiro isolamos o valor absoluto: } -|x - 2| + 6 = 2 \;\Rightarrow\; -|x - 2| = -4
\]





\[
\text{Multiplicando ambos os lados por } -1, \text{ obtemos } |x - 2| = 4
\]





\[
\text{Como o valor absoluto de uma expressão é igual a 4, isso significa que } x - 2 = 4 \text{ ou } x - 2 = -4
\]





\[
\text{No primeiro caso: } x - 2 = 4 \;\Rightarrow\; x = 6
\]





\[
\text{No segundo caso: } x - 2 = -4 \;\Rightarrow\; x = -2
\]



% Solução final destacada


\[
\boxed{x = -2 \quad \text{ou} \quad x = 6}
\]
